// Parser les coefficients du CSV function parseCoefficients(csvText) { coefficients = {}; featureTypes = {}; featureDirections = {}; const lines = csvText.trim().split('\n'); // Parser chaque ligne (format basé sur votre exemple) for (let i = 0; i < lines.length; i++) { // Skip header si présent if (i === 0 && lines[i].toLowerCase().includes('feature')) continue; // Nettoyer la ligne const line = lines[i].trim(); if (!line) continue; // Parser le CSV en tenant compte des guillemets const parts = parseCSVLine(line); if (parts.length >= 2) { const feature = parts[0].trim().replace(/^"|"$/g, ''); // Le coefficient est toujours en position 1 let coeffValue = parseFloat(parts[1].replace(/^"|"$/g, '').replace(',', '.')); if (!isNaN(coeffValue)) { coefficients[feature] = coeffValue; // Déterminer le type (colonne 3: "Type") if (parts.length > 3) { const typeValue = parts[3].trim().replace(/^"|"$/g, ''); featureTypes[feature] = typeValue.includes('Clinique') ? 'Clinique' : 'Gène'; } else { featureTypes[feature] = feature.startsWith('ENSG') ? 'Gène' : 'Clinique'; } // Déterminer la direction (colonne 4: "Effect_Type") if (parts.length > 4) { const effectValue = parts[4].trim().replace(/^"|"$/g, ''); featureDirections[feature] = effectValue.includes('Défavorable') ? 'Défavorable' : 'Favorable'; } else { featureDirections[feature] = coeffValue > 0 ? 'Favorable' : 'Défavorable'; } } } }
